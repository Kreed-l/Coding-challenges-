var capitals = function (word) {
	let arr = []  //empty array to fill with index
  //if current letter is same as letter changed to caps then its pushed 
  for(let i=0; i<word.length; i++){
    word[i] === word[i].toUpperCase() ? arr.push(i) : ''
  }
  return arr
};
